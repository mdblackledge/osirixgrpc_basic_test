!$certauth = certauth

!function $keypairfname($name)
!return $name.privkey.pem\n$name.pubcert.pem
!endfunction

[$keypairfname($certauth)]
[$keypairfname($server)]
[server.ipaddresses.json]
}

node "Client"{
[$keypairfname($client)]
}

[$keypairfname($certauth)] --> [$keypairfname($server)] : signs
[$keypairfname($certauth)] --> [$keypairfname($client)] : signs
